# 常见排序算法

## 归并排序

```C++
#include<bits/stdc++.h>
using namespace std;

void merge(vector<int> &nums, int s, int m, int e) {
	vector<int> res(e - s + 1);
	int i = s, j = m + 1, k = 0;
	while (i <= m && j <= e) {
		if (nums[i] <= nums[j]) {
			res[k++] = nums[i++];
		} else {
			res[k++] = nums[j++];
		}
	}
	while (i <= m) res[k++] = nums[i++];
	while (j <= e) res[k++] = nums[j++];
	for (int i = s; i <= e; i++)
		nums[i] = res[i - s];
}

void mergeSort(vector<int> &nums, int s, int e) {
	if (s >= e) return;
	int m = (s + e) / 2;
	mergeSort(nums, s, m);   //递归的将L[s...m]归并为有序的L[s...m]
	mergeSort(nums, m + 1, e);   //递归的将L[m+1...e]归并为有序的L[m+1...e]
	merge(nums, s, m, e);   //将有序的L[s...m]和L[m+1...e]归并到L[s...e]
}

int main() {
	int a[] = {1, 2, 3, 3, 2, 1, 0};
	int size = sizeof(a) / sizeof(int);
	vector<int> nums(a, a + size);	
	mergeSort(nums, 0, (int)nums.size() - 1);
	for (int num: nums)
		cout << num << " ";
	return 0;
}
```
